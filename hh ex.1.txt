 /*
 * Функция возвращает остаток от деления любого числа на 10
 */
function my_bcmod( $x )
{
	$take = 5;
	$mod = '';
	do
	{
		$a = (int)$mod.substr( $x, 0, $take );
		$x = substr( $x, $take );
		$mod = $a % 10;
	}
	while ( strlen($x) );
	return (int)$mod;
}
/*
 * Функция возвращает перевернутое число 
 * любого количества знаков.
 * */
function rotateNum( $num ){
	$len = strlen($num);
	for($i = 0; $i < $len; $i++){
		$residue = my_bcmod($num);
		$num = bcdiv($num, 10);
		(string)$numRotate = $numRotate . $residue;
	}
	return (string)$numRotate;
}
